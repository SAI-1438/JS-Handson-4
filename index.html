Q10.Min and Max

function arrayMin(arr) {
        let minvalue= arr[0];
        for(i=0;i < arr.length;i++){
          if(arr[i] < minvalue){
             minvalue=arr[i];
          }
        }
      return minvalue;
      }
      
      function arrayMax(arr) {
       let maxvalue=arr[0];
       for(i=0;i < arr.length;i++){
         if(arr[i]>maxvalue){
           maxvalue=arr[i]
         }
       }
      return maxvalue;
      }

Q11.Birthday Game

function Birthday_Game(arr,D ,M) {
        let N=arr.length;
       let count=0;
       for(let i=0;i < N-M+1;i++){
         let sum=0;
         for(let j=i;j < M+i;j++){
           sum+=arr[j];
         }
         if(sum===D)
         count++;
       }
       return count;
       
       }
